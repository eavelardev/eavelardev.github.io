def contains(sentence):
    return catalog_data['name'].str.lower().str.contains(sentence)

def check_sentences(sentences):
    check = contains(sentences[0])

    for sentence in sentences[1:]:
        check |= contains(sentence)

    return check

valid_sentences = [
    'machine learning',
    'ml',
